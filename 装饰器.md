#   装饰器
---
### 装饰器=高阶函数+函数嵌套+函数闭包
#### 高阶函数的定义
-   函数接收的参数是一个函数名
-   函数都返回值是一个函数名
-   满足上述条件任意一个，都可称为高阶函数
#### 高阶函数示例
```
import time

def foo():
    time.sleep(2)
    print('hello ali')

def test(func):
    start_time = time.time()
    func()
    end_time = time.time()
    print('运行时间为%s' %(end_time - start_time))

test(foo)
```
#### 函数嵌套
-   函数嵌套就是函数里边定义了一个函数
```
def ali(name):
    print('来自ali%s' %name)
    def ayisha():
        print('来自ayisah%s' %name)
        def haha():
            print('来自haha%s' %name)
        haha()
    ayisha()
ali('test')
```
#### 函数闭包
-   函数闭包的表现形式和变量的作用域一样
```
def ali(name):
    def ayisha():
        def haha():
            print('来自haha%s' %name)
        haha()
    ayisha()
ali('test')
```
#### 函数闭包装饰器实现
-   @符号+装饰器函数名 将@符号放在需要被装饰的函数之前，就完成了对改函数的装饰
```
装饰前
import time

def timerp(fun):
    def war():
        start_time = time.time()
        fun()
        stop_time = time.time()
        print('ali函数的运行时间是%s' %(stop_time - start_time))
    return war


def ali():
    time.sleep(2)
    print('from ali')

ali = timerp(ali)
ali()
```
```
装饰后
import time

def timerp(fun):
    def war():
        start_time = time.time()
        fun()
        stop_time = time.time()
        print('ali函数的运行时间是%s' %(stop_time - start_time))
    return war

@timerp
def ali():

ali()
```
#### 函数闭包装饰器加上返回值
```
import time

def timerp(fun):
    def war():
        start_time = time.time()
        res=fun()
        stop_time = time.time()
        print('ali函数的运行时间是%s' %(stop_time - start_time))
        return res
    return war

@timerp
def ali():
    time.sleep(2)
    print('from ali')
    return '这是ali的返回值'

res = ali()
print(res)
```
#### 函数闭包装饰器加参数
```
import time

def used_time(func):
    def used(*args,**kwargs):
        start_time = time.time()
        res = func(*args,**kwargs)
        stop_time = time.time()
        print('cmd执行时间是%s' %(stop_time - start_time))
        return res
    return used

@used_time
def cmd(name,commond):
    print('cmd is %s admin is %s' %(commond,name))
    return 'for cmd return'

x = cmd('ali','ll')
y = cmd('ali','cd')
print(x,y)
```
#### 装饰器加上验证功能
```
def auth_sa(func):
    def wrap(*args,**kwargs):
        name = input('username:').strip()
        passwd = input('userpasswd').strip()
        if name == 'ali' and passwd == '123':
            res = func(*args,**kwargs)
            return res
        else:
            print('enter error')

#### 函数闭包带参数的装饰器
```
user_list = [
    {'name':'ali','passwd':'ali'},
    {'name':'haha','passwd':'haha'}
]

user_init = {'username':None,'login':False}

def auth_type(type='ldap'):
    def auth_sa(func):
        def wrap(*args,**kwargs):
            print('type is',type)
            if type == 'ldap':
                if user_init['username'] and user_init['login']:
                    res = func(*args,**kwargs)
                    return res
                username = input('username:').strip()
                userpasswd = input('userpasswd:').strip()
                for user_dic in user_list:
                    if username == user_dic['name'] and userpasswd == user_dic['passwd']:
                        user_init['username']=username
                        user_init['login']=True
                        res = func(*args,**kwargs)
                        return res
                else:
                    print('auth erro')
            else:
                print('不会玩')
        return wrap
    return auth_sa

@auth_type(type='ldap')
def sa():
    print('欢迎来到运维部')
